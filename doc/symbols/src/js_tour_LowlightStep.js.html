1 /**
  2  * Generates one help step
  3  * Lowlight an element
  4  *
  5  * @author Georg Limbach <georf@dev.mgvmedia.com>
  6  *
  7  * @constructor
  8  * @param {String} element jquery selector string
  9  */
 10 function LowlightStep(element) {
 11 	var $this = this;
 12 	this._element = element;
 13 
 14 	/**
 15 	 * run method
 16 	 * @type Function
 17 	 */
 18 	this.run = function() {
 19 		$(element).css('z-index', 1);
 20 	}
 21 
 22 	/**
 23 	 * time for this step
 24 	 * @type int
 25 	 */
 26 	this.time = 100;
 27 }
